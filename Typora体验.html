#include <bits/stdc++.h>
#define N 50020
#define Side(x) for(int i=p[x];i;i=nt[i])
using namespace std;
int a[N],c[N],tot,id,top,e[N<<1],nt[N<<1],p[N],SCC,dfn[N],low[N],bel[N],stk[N],f[N],x[N<<1],y[N<<1];
bool vis[N];
inline void add(int x,int y)
{
    e[++tot]=y,nt[tot]=p[x],p[x]=tot;
}
void Tarjan(int x)
{
    low[x]=dfn[x]=++id,vis[x]=true,stk[++top]=x;
    Side(x)
        if(!dfn[e[i]])
            Tarjan(e[i]),low[x]=min(low[x],low[e[i]]);
        else if(vis[e[i]])low[x]=min(low[x],dfn[e[i]]);
    if(low[x]==dfn[x])
    {
        bel[x]=++SCC,vis[x]=false,c[SCC]+=a[x];
        while(stk[top]!=x)
            bel[stk[top]]=SCC,c[SCC]+=a[stk[top]],vis[stk[top--]]=false;
        top--;
    }
}
int dfs(int x)
{
    if(f[x])return f[x];
    f[x]=c[x];
    int mx=0;
    Side(x)
        mx=max(mx,dfs(e[i]));
    return f[x]+=mx;
}
int main()
{
    int n,m,xi,yi,ans=0;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(int i=1;i<=m;i++)
        scanf("%d%d",&x[i],&y[i]),add(x[i],y[i]);
    for(int i=1;i<=n;i++)
        if(!bel[i])Tarjan(i);
    memset(p,0,sizeof(p));
    for(int s=1;s<=m;s++)
        if(bel[x[s]]!=bel[y[s]])add(bel[x[s]],bel[y[s]]);
    for(int i=1;i<=SCC;i++)
        if(!f[i])ans=max(ans,dfs(i));
    cout<<ans;
}
